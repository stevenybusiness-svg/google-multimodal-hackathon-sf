**7. drawEdge(ctx, edge, time):**
Per UI-SPEC:
- Get source and target pixel positions via `px()`

**8. drawNode(ctx, node, time):**

**9. Particle system:**
Module-level array `const particles = [];`
```javascript
function spawnParticles(edgeKey, color) {
  const edge = EDGES.find(e => `${e.from}->${e.to}` === edgeKey);
  if (!edge) return;
  const count = 3 + Math.floor(Math.random() * 3); // 3-5 particles
  for (let i = 0; i < count; i++) {
    particles.push({
      edge: edgeKey,
      t: -i * 0.05, // stagger start (50ms equivalent in t-offset)
      color,
      speed: 0.008 + Math.random() * 0.004, // ~2s traversal
    });
  }
}
```

`function updateAndDrawParticles(ctx, dt, time):`
- For each particle, advance `t` by `speed` (speed is per-frame at 60fps, so multiply by `dt * 60` to be frame-rate independent)
- Skip drawing if `t < 0` (staggered start)
- Remove if `t >= 1`
- Get position via quadratic Bezier: source -> control point -> target
- Draw: `ctx.arc(pos.x, pos.y, 3, 0, Math.PI * 2)`, fillStyle = particle.color, shadowColor = particle.color, shadowBlur = 8
- Reset shadowBlur after each particle

**10. Bezier point helper:**
```javascript
function bezierPoint(x0, y0, cx, cy, x1, y1, t) {
  const u = 1 - t;
  return {
    x: u * u * x0 + 2 * u * t * cx + t * t * x1,
    y: u * u * y0 + 2 * u * t * cy + t * t * y1,
  };
}
```

**11. Animation loop:**
```javascript
let animationId = null;
let lastTime = 0;

function animate(currentTime) {
  animationId = requestAnimationFrame(animate);
  const dt = Math.min((currentTime - lastTime) / 1000, 0.1); // clamp to 100ms
  lastTime = currentTime;
  const time = currentTime / 1000;

  ctx.clearRect(0, 0, canvasW, canvasH);

  // Layer 1: Edges (back)
  for (const edge of EDGES) drawEdge(ctx, edge, time);
  // Layer 2: Particles (middle)
  updateAndDrawParticles(ctx, dt, time);
  // Layer 3: Nodes (front)
  for (const node of NODES) drawNode(ctx, node, time);
}
```

**12. startAnimation() / stopAnimation():**
```javascript
function startAnimation() {
  if (animationId) return;
  initCanvas();
  lastTime = performance.now();
  animationId = requestAnimationFrame(animate);
}
function stopAnimation() {
  if (animationId) {
    cancelAnimationFrame(animationId);
    animationId = null;
  }
}
```

**13. handlePipelineEvent(data) — THE CORE WIRING:**
Maps backend events to node activations, edge activations, particle spawns, and stat updates.

```javascript
const EVENT_MAP = {
  'stt_start':           { nodes: [{id:'mic', dur:1.0}], edges: ['mic->stt'] },
  'stt_result':          { nodes: [{id:'stt', dur:1.5}], edges: ['stt->gemini'] },
  'understanding_start': { nodes: [{id:'gemini', dur:2.0}], edges: [] },
  'understanding_result':{ nodes: [{id:'gemini', dur:2.0}], edges: ['gemini->dispatcher'] },
  'vision_result':       { nodes: [{id:'camera', dur:1.0}, {id:'vision', dur:1.5}], edges: ['camera->vision', 'vision->gemini'] },
  'action_dispatched':   { nodes: [{id:'dispatcher', dur:1.0}], edges: [] }, // edge added dynamically
};

function handlePipelineEvent(data) {
  if (!data || !data.event) return;
  const now = performance.now() / 1000;
  const mapping = EVENT_MAP[data.event];
  if (!mapping) return;

  // Hide empty state on first event
  const emptyState = window.MeetingAgent.core.dom.pipelineEmptyState;
  if (emptyState) emptyState.style.display = 'none';

  // Activate nodes
  for (const { id, dur } of mapping.nodes) {
    const node = nodeById(id);
    if (node) node.activeUntil = now + dur;
  }

  // Activate edges + spawn particles
  for (const ek of mapping.edges) {
    const edge = EDGES.find(e => `${e.from}->${e.to}` === ek);
    if (edge) {
      edge.activeUntil = now + 2.0;
      const srcNode = nodeById(edge.from);
      spawnParticles(ek, srcNode ? srcNode.color : '#57abff');
    }
  }

  // Handle action_dispatched: dynamic edge based on action_type
  if (data.event === 'action_dispatched' && data.action_type) {
    const targetMap = { slack: 'slack', calendar: 'calendar', task: 'tasklog', document: 'slack' };
    const target = targetMap[data.action_type] || 'tasklog';
    const ek = `dispatcher->${target}`;
    const edge = EDGES.find(e => `${e.from}->${e.to}` === ek);
    if (edge) {
      edge.activeUntil = now + 2.0;
      spawnParticles(ek, '#57abff');
    }
    const targetNode = nodeById(target);
    if (targetNode) {
      targetNode.activeUntil = now + 1.0;
      // Update target node count stat
      targetNode.stats.count = (targetNode.stats.count || 0) + 1;
    }
    // Update dispatcher fired count
    const dispNode = nodeById('dispatcher');
    if (dispNode) dispNode.stats.fired = (dispNode.stats.fired || 0) + 1;
  }

  // Update stats from event data
  if (data.event === 'stt_result' && data.stats) {
    const sttNode = nodeById('stt');
    if (sttNode) sttNode.stats.words = (sttNode.stats.words || 0) + (data.stats.words || 0);
  }
  if (data.event === 'understanding_result' && data.stats) {
    const gemNode = nodeById('gemini');
    if (gemNode) gemNode.stats.actions = (gemNode.stats.actions || 0) + (data.stats.actions || 0);
  }
  if (data.event === 'vision_result' && data.stats) {
    const visNode = nodeById('vision');
    if (visNode) {
      visNode.stats.sentiment = data.stats.sentiment || '';
      visNode.stats.score = data.stats.score || '';
    }
  }

  // Start animation if not running
  startAnimation();
}
```
